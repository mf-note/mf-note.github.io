#### 父组件

```vue
<template>
  <div class="parent">
    <Child>
      <span #slot1="{ item, index }">{{ index }}</span>

      <span #slot2="{ item, index }">{{ index }}</span>
    </Child>
  </div>
</template>

<script>
import Child from './Child.vue'
export default {
  name: "parent",
  data() {
    return {
      slots: ['slot1', 'slot2', 'slot3'],
      item: {}
    }
  }
}
</script>
```

#### 子组件

```vue
<template>
  <div class="child">
    <Sun>
      <template v-for="(index, name) in $scopedSlots" v-slot:[name]="data">
        <slot :name="name" v-bind="data"></slot>
      </template>
    </Sun>
  </div>
</template>

<script>
import Sun from './Sun.vue'
export default {
  name: "child",
  data() {
    return {
      slots: ['slot1', 'slot2', 'slot3'],
      item: {}
    }
  }
}
</script>
```

#### 孙组件

```vue
<template>
  <div class="parent">
    <template v-for="(name, index) in slots">
      <slot :name="name" :item="item" :index="index"></slot>
    </template>
  </div>
</template>

<script>
export default {
  name: "sun",
  data() {
    return {
      slots: ['slot1', 'slot2', 'slot3'],
      item: {}
    }
  }
}
</script>
```
